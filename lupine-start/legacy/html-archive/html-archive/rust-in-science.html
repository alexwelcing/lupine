Legacy C++ 
// Manual memory management
void compute_forces(Material* m) {
    for (int i = 0; i <= m->size; i++) {
        // Buffer overflow risk
        double f = calculate(m->atoms[i]);
        // Manual pointer arithmetic
        *(m->results + i) = f;
    }
    // Did we free m? Double free risk
    delete m;
}
Safe Rust 
// Ownership and Iterators
fn compute_forces(m: Material) -> Vec<f64> {
    m.atoms.par_iter()
        .map(|atom| {
            // Guaranteed memory safety
            calculate(atom)
        })
        // Automatic cleanup
        .collect()
}
